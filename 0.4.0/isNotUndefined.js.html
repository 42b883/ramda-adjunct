const { complement } = require('ramda');

/**
 * Checks if input `value` is complement `undefined`

 * @func isNotUndefined
 * @memberOf RA
 * @since v0.0.1
 * @category Type
 * @sig * -> Boolean
 * @param {*} val The value to test
 * @return {Boolean}
 * @see {@link RA.isUndefined|isUndefined}
 * @example
 *
 * RA.isNotUndefined(1); //=> true
 * RA.isNotUndefined(undefined); //=> false
